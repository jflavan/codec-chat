# SFU TLS Termination Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Encrypt the SFU control API traffic by adding nginx TLS termination with automated Let's Encrypt certificates on the voice VM.

**Architecture:** Nginx reverse proxy on the voice VM terminates TLS on port 443 and forwards to the SFU on `127.0.0.1:3001`. Certbot handles Let's Encrypt certificate provisioning and renewal. An Azure DNS A record points `sfu.codec-chat.com` at the VM's static IP. The SFU binds to localhost only.

**Tech Stack:** Bicep (IaC), nginx, certbot/Let's Encrypt, GitHub Actions CD

**Spec:** `docs/superpowers/specs/2026-03-10-sfu-tls-design.md`

- [ ] **Step 2: Replace the cloudInit variable**

Replace the entire `var cloudInit = '''...'''` block (lines 195-212):

```bicep
var cloudInit = '''
#cloud-config

package_update: true
package_upgrade: false

packages:
  - docker.io
  - docker-compose-v2
  - curl
  - jq
  - nginx
  - certbot
  - python3-certbot-nginx

write_files:
  - path: /etc/nginx/sites-available/sfu-proxy
    content: |
      server {
          listen 80 default_server;
          server_name _;
          # Serve only ACME challenges; reject everything else.
          location /.well-known/acme-challenge/ {
              root /var/www/html;
          }
          location / {
              return 444;
          }
      }
      server {
          listen 443 ssl default_server;
          server_name ${sfuDomainName};

          # Placeholder self-signed cert — certbot replaces this on first run.
          ssl_certificate /etc/ssl/certs/ssl-cert-snakeoil.pem;
          ssl_certificate_key /etc/ssl/private/ssl-cert-snakeoil.key;

          ssl_protocols TLSv1.2 TLSv1.3;
          ssl_prefer_server_ciphers on;

          location / {
              proxy_pass http://127.0.0.1:3001;
              proxy_set_header Host $host;
              proxy_set_header X-Real-IP $remote_addr;
              proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
              proxy_set_header X-Forwarded-Proto $scheme;
              proxy_read_timeout 30s;
              proxy_connect_timeout 5s;
          }

          location /health {
              proxy_pass http://127.0.0.1:3001/health;
          }
      }

runcmd:
  - systemctl enable docker
  - systemctl start docker
  - mkdir -p /opt/voice
  - chown azureuser:azureuser /opt/voice
  # Configure nginx: enable the SFU proxy site, disable the default site.
  - ln -sf /etc/nginx/sites-available/sfu-proxy /etc/nginx/sites-enabled/sfu-proxy
  - rm -f /etc/nginx/sites-enabled/default
  - systemctl enable nginx
  - systemctl start nginx
  # Obtain a Let's Encrypt certificate. The --nginx plugin updates the config in place.
  # Uses --non-interactive so cloud-init does not hang waiting for input.
  - certbot --nginx -d ${sfuDomainName} --non-interactive --agree-tos --email ${certbotEmail} --redirect
  - systemctl reload nginx
'''
```

Note: `${sfuDomainName}` and `${certbotEmail}` in the cloud-init string are Bicep interpolation variables — they get substituted at deployment time, not by the VM. However, because cloud-init is inside a Bicep multi-line string literal (`'''`), Bicep does NOT interpolate variables. We need to use string concatenation instead.

- [ ] **Step 3: Convert cloud-init to use Bicep string concatenation**

Replace the `var cloudInit` with a concatenation approach that injects the domain name and email:

```bicep
var cloudInit = format('''
#cloud-config

package_update: true
package_upgrade: false

packages:
  - docker.io
  - docker-compose-v2
  - curl
  - jq
  - nginx
  - certbot
  - python3-certbot-nginx

write_files:
  - path: /etc/nginx/sites-available/sfu-proxy
    content: |
      server {{
          listen 80 default_server;
          server_name _;
          location /.well-known/acme-challenge/ {{
              root /var/www/html;
          }}
          location / {{
              return 444;
          }}
      }}
      server {{
          listen 443 ssl default_server;
          server_name {0};

          ssl_certificate /etc/ssl/certs/ssl-cert-snakeoil.pem;
          ssl_certificate_key /etc/ssl/private/ssl-cert-snakeoil.key;

          ssl_protocols TLSv1.2 TLSv1.3;
          ssl_prefer_server_ciphers on;

          location / {{
              proxy_pass http://127.0.0.1:3001;
              proxy_set_header Host $host;
              proxy_set_header X-Real-IP $remote_addr;
              proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
              proxy_set_header X-Forwarded-Proto $scheme;
              proxy_read_timeout 30s;
              proxy_connect_timeout 5s;
          }}

          location /health {{
              proxy_pass http://127.0.0.1:3001/health;
          }}
      }}

runcmd:
  - systemctl enable docker
  - systemctl start docker
  - mkdir -p /opt/voice
  - chown azureuser:azureuser /opt/voice
  - ln -sf /etc/nginx/sites-available/sfu-proxy /etc/nginx/sites-enabled/sfu-proxy
  - rm -f /etc/nginx/sites-enabled/default
  - systemctl enable nginx
  - systemctl start nginx
  - certbot --nginx -d {0} --non-interactive --agree-tos --email {1} --redirect
  - systemctl reload nginx
''', sfuDomainName, certbotEmail)
```

Important: Bicep `format()` uses `{{` and `}}` to produce literal braces in the output. All nginx config braces must be doubled.

```yaml
      - name: Ensure nginx TLS proxy is configured
        run: |
          az vm run-command invoke \
            --resource-group ${{ env.RESOURCE_GROUP }} \
            --name ${{ env.VOICE_VM_NAME }} \
            --command-id RunShellScript \
            --scripts "
              set -e
              if command -v nginx &>/dev/null && [ -f /etc/letsencrypt/live/sfu.codec-chat.com/fullchain.pem ]; then
                echo 'nginx + certbot already configured, reloading'
                systemctl reload nginx
                exit 0
              fi

              echo 'Installing nginx and certbot...'
              apt-get update -qq
              apt-get install -y -qq nginx certbot python3-certbot-nginx

              # Write nginx proxy config
              cat > /etc/nginx/sites-available/sfu-proxy <<'NGINX'
              server {
                  listen 80 default_server;
                  server_name _;
                  location /.well-known/acme-challenge/ {
                      root /var/www/html;
                  }
                  location / {
                      return 444;
                  }
              }
              server {
                  listen 443 ssl default_server;
                  server_name sfu.codec-chat.com;

                  ssl_certificate /etc/ssl/certs/ssl-cert-snakeoil.pem;
                  ssl_certificate_key /etc/ssl/private/ssl-cert-snakeoil.key;
                  ssl_protocols TLSv1.2 TLSv1.3;
                  ssl_prefer_server_ciphers on;

                  location / {
                      proxy_pass http://127.0.0.1:3001;
                      proxy_set_header Host \\\$host;
                      proxy_set_header X-Real-IP \\\$remote_addr;
                      proxy_set_header X-Forwarded-For \\\$proxy_add_x_forwarded_for;
                      proxy_set_header X-Forwarded-Proto \\\$scheme;
                      proxy_read_timeout 30s;
                      proxy_connect_timeout 5s;
                  }

                  location /health {
                      proxy_pass http://127.0.0.1:3001/health;
                  }
              }
          NGINX
              ln -sf /etc/nginx/sites-available/sfu-proxy /etc/nginx/sites-enabled/sfu-proxy
              rm -f /etc/nginx/sites-enabled/default
              systemctl enable nginx
              systemctl restart nginx

              # Obtain Let's Encrypt cert
              certbot --nginx -d sfu.codec-chat.com \
                --non-interactive --agree-tos \
                --email ${{ secrets.CERTBOT_EMAIL }} \
                --redirect

              systemctl reload nginx
              echo 'nginx TLS proxy configured successfully'
            "
```
